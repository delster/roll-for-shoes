import RuleWrap from './rule-wrap.js'
import RuleList from './rule-list.js'
import Rule from './rule.js'

<RuleWrap>

  # Rules

  <RuleList>
    <Rule> Say what you do and roll a number of d6s. </Rule>
    <Rule> If the sum of your roll is higher than the opposing roll, the thing you wanted to happen, happens. </Rule>
    <Rule> The number of d6s you roll is determined by the level of the skill you have. </Rule>
    <Rule> At start, you have only one skill: Do anything 1. </Rule>
    <Rule> If you roll all sixes, you get a new skill specific to the action, one level higher than the one you used. </Rule>
    <Rule> For every roll you fail, you get 1 XP. </Rule>
    <Rule> XP can be used to change a die into a 6 for advancement purposes only. </Rule>
  </RuleList>

</RuleWrap>
